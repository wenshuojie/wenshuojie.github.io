-
```
var a = 123;
function fun(){
    console.log("a="+a);//函数作用域中没有a，向上寻找
}
fun();

a=123
```

-
```
var a = 123;
function fun() {
    console.log("a="+a);//函数中用var声明了a，声明提前但不赋值
    var a = 456;
}
fun();
console.log("a="+a);//全局中有a=123

a=undefined
a=123
```

-
```
var a = 123;
function fun(){
    console.log("a="+a);//在函数作用域中，a没有用var声明，为全局变量，则函数未声明a，向上寻找
    a = 456;//全局a被改变
}
fun();
console.log("a="+a);

a=123
a=456
```

-
```
var a = 123;
function fun(a){
    //函数有形参a，相当于在函数体内部声明了
    //var = a;
    console.log("a="+a);//a在函数中声明了但是没有赋值
    a = 456;//改变函数作用域中的a
}
fun();
console.log("a="+a);//全局有a

a=undefined
a=123
```

-
```
var a = 123;
function fun (a){
    //函数有形参a，相当于在函数体内部声明了
    //var = a;
    console.log("a="+a);
    a = 456;//改变函数内的a
}
fun(123);//传入实参
console.log("a="+a);//全局a

a=123
a=123
```  

-
```
var a = "全局";
var fun = function() {
    var a = "fun内部";
    function fun2() {
        //fun2中使用var声明了a，相当于在fun2开始之前执行：var a;
        console.log(a);//只声明但未赋值
        a = "hhh";//将fun2中的a赋值为"hhh"
        console.log(a);
        var a = "kkk";
        console.log(a);
    }
    console.log("fun2执行前="+a);
    fun2();
    console.log("fun2执行后="+a);
}
fun()

fun2执行前=fun内部
undefined
hhh
kkk
fun2执行前=fun内部
```

-
```
var a = "全局";
var fun = function() {
    var a = "fun内部";
    function fun2() {
        console.log(a);
        a = "hhh";
        console.log(a);
        a = "kkk";
        console.log(a);
    }
    console.log("执行fun2之前fun内部的a="+a);
    fun2();
    console.log("执行fun2之前fun内部的a="+a);
}
fun()

执行fun2之前fun内部的a=fun内部
fun内部
hhh
kkk
执行fun2之前fun内部的a=kkk
```

-
```
var a = "全局";
var fun = function() {
    function fun2() {
        console.log(a);
        a = "hhh";
        console.log(a);
        a = "kkk";
        console.log(a);
        d = "fun2内部创建的d"//没有用var声明，函数中没有；全局也没有，添加到全局
    }
    console.log("fun2执行前="+a);
    fun2();
    console.log("fun2执行后="+a);
}
fun();
console.log(d);

fun2执行前=全局
全局
hhh
kkk
fun2执行后=kkk
fun2内部创建的d
```
